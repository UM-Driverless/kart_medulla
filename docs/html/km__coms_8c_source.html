1/******************************************************************************
2 * @file km_coms.c
3 * @brief Implementation of the KM_COMS communication library for ESP32.
4 *
5 * This file implements the KM_COMS library, providing UART-based communication
6 * between an ESP32 and an NVIDIA Orin (or other host system). The library handles:
7 * - Initialization and configuration of the UART peripheral
8 * - Sending and receiving framed messages
9 * - CRC8 validation for data integrity
10 * - Processing incoming payloads and updating shared application objects
11 * - FreeRTOS task integration for periodic message handling
12 *
13 * --------------------------------------------------------------------------
14 * PROTOCOL DETAILS
15 * --------------------------------------------------------------------------
16 * Each message follows a custom framing format:
17 *
18 * | SOF | LEN | TYPE | PAYLOAD | CRC |
19 *
20 * Fields:
21 * - SOF : Start-of-frame byte (0xAA)
22 * - LEN : 1-byte payload length
23 * - TYPE : 1-byte message type (see message_type_t in km_coms.h)
24 * - PAYLOAD: N bytes of data up to 251 bytes, content depends on message type
25 * - CRC : 1-byte CRC8 (XOR-based) over LEN, TYPE, and PAYLOAD
26 *
27 * Total size of message is 255 bytes
28 *
29 * The `message_type_t` enum defines all possible message types. Conceptually:
30 * - ESP32 → Orin messages contain telemetry data such as current speed, throttle,
31 * braking, steering, mission state, machine state, and shutdown status.
32 * - Orin → ESP32 messages carry commands such as target throttle, braking,
33 * steering, mission state, machine state, and heartbeat.
34 *
35 * Using the enum ensures type safety and consistency when sending or processing messages.
36 *
37 * The CRC ensures the integrity of the message. Invalid messages are discarded.
38 *
39 * The library uses internal buffers and a FreeRTOS mutex to provide thread-safe
40 * access to shared objects and message queues.
41 *
42 * --------------------------------------------------------------------------
43 * FREE RTOS INTEGRATION
44 * --------------------------------------------------------------------------
45 * Periodic tasks should call km_coms_ReceiveMsg() and KM_COMS_ProccessMsgs() to
46 * read bytes from UART, assemble messages, validate CRC, and process payloads.
47 * KM_COMS_SendMsg() can be used to transmit messages safely from tasks.
48 *
49 * Author: Adrian Navarredonda Arizaleta
50 * Date: 14-02-2026
51 * Version: 1.0
52 *****************************************************************************/

80esp_err_t KM_COMS_Init(uart_port_t uart_port) {
81 km_coms_uart = uart_port;
82
83 km_coms_mutex = xSemaphoreCreateMutex();
84 if(km_coms_mutex == NULL)
85 return ESP_ERR_NO_MEM;
86
87 // Instala el driver UART con buffers TX/RX
88 uart_driver_install(km_coms_uart, BUF_SIZE_RX, BUF_SIZE_TX, 0, NULL, 0);
89 uart_param_config(km_coms_uart, &uart_config);
90
91 // Asigna pines según el puerto
92 if(km_coms_uart == UART_NUM_2) {
93 uart_set_pin(km_coms_uart, PIN_ORIN_UART_TX, PIN_ORIN_UART_RX,
94 UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
95 } else {
96 uart_set_pin(km_coms_uart, PIN_USB_UART_TX, PIN_USB_UART_RX,
97 UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
98 }
99
100 return ESP_OK;
101}

103int KM_COMS_SendMsg(message_type_t type, uint8_t *payload, uint8_t len) {
104 km_coms_msg msg;
105 uint8_t frame[KM_COMS_MSG_MAX_LEN - 1];
106 size_t total_sent = 0;
107 int sent, attempts = 0;
108
109 if(len > KM_COMS_MSG_MAX_LEN)
110 return -1;
111
112 msg.len = (uint8_t)len;
113 msg.type = (uint8_t)type;
114
115 memcpy(msg.payload, payload, len);
116 msg.crc = KM_COMS_crc8(msg.len, msg.type, msg.payload); // LEN + TYPE + PAYLOAD
117
118 // Armar frame
119 frame[0] = (uint8_t)KM_COMS_SOM;
120 frame[1] = (uint8_t)msg.len;
121 frame[2] = (uint8_t)msg.type;
122 memcpy(&frame[3], msg.payload, msg.len);
123 frame[3 + msg.len] = (uint8_t)msg.crc;
124
125 // Enviar mensaje, se intenta 5 veces
126 while(total_sent < len+4 && attempts < 5) {
127 sent = uart_write_bytes(km_coms_uart, frame + total_sent, (len + 4) - total_sent);
128 total_sent += sent;
129
130 if(sent < (len - total_sent)) {
131 // buffer lleno, espera a que se vacíe
132 attempts++;
133 vTaskDelay(5 / portTICK_PERIOD_MS);
134 }
135 }
136
137 // NO se ha podido enviar el mensaje correctamente
138 if(attempts >= 5 || total_sent < 4 + len){
139 ESP_LOGE("KM_coms", "El msg no se ha enviado, bytes enviados: %d, bytes que habia que enviar: %d", total_sent, 4+len);
140 return 0;
141 }
142
143 // Se ha enviado el mensaje correctamente
144 return 1;
145}

148void km_coms_ReceiveMsg(void) {
149 uint8_t uart_chunk[KM_COMS_RX_CHUNK];
150 size_t len_read = 0;
151 uint8_t bytes2read = 0;
152
153 // 1. Verificar cuantos bytes hay en el buffer UART
154 size_t uart_len = 0;
155 uart_get_buffered_data_len(km_coms_uart, &uart_len);
156 if(uart_len == 0)
157 return;
158
159 // 2. Leer hasta KM_COMS_RX_CHUNK bytes de la UART
160 if(uart_len > KM_COMS_RX_CHUNK)
161 bytes2read = KM_COMS_RX_CHUNK;
162 else
163 bytes2read = uart_len;
164
165 len_read = uart_read_bytes(km_coms_uart, uart_chunk, bytes2read, 0);
166 if(len_read == 0)
167 return;
168
169 if(xSemaphoreTake(km_coms_mutex, pdMS_TO_TICKS(KM_COMS_WAIT_SEM_AVAILABLE)) == pdTRUE) {
170 // 3. Copiar bytes al buffer interno
171 if(rx_buffer_len + len_read > sizeof(rx_buffer)) {
172 // Overflow, reiniciar buffer
173 rx_buffer_len = 0;
174 }
175 memcpy(rx_buffer + rx_buffer_len, uart_chunk, len_read);
176 rx_buffer_len += len_read;
177 xSemaphoreGive(km_coms_mutex);
178 }
179}

181void KM_COMS_ProccessMsgs(void) {
182 size_t processed = 0;
183
184 if(xSemaphoreTake(km_coms_mutex, pdMS_TO_TICKS(KM_COMS_WAIT_SEM_AVAILABLE)) != pdTRUE)
185 return;
186
187 while(rx_buffer_len - processed >= 4) { // Mínimo: SOM + LEN + TYPE + CRC
188 if(rx_buffer[processed] != KM_COMS_SOM) {
189 processed++;
190 continue;
191 }
192
193 uint8_t payload_len = rx_buffer[processed + 1];
194 size_t total_len = 4 + payload_len; // SOM + LEN + TYPE + PAYLOAD + CRC
195
196 if(rx_buffer_len - processed < total_len)
197 break; // Mensaje incompleto, esperar más bytes
198
199 // Construir mensaje
200 km_coms_msg msg;
201 msg.len = payload_len;
202 msg.type = rx_buffer[processed + 2];
203 memcpy(msg.payload, rx_buffer + processed + 3, payload_len);
204 msg.crc = rx_buffer[processed + 3 + payload_len];
205
206 // Verificar CRC
207 uint8_t crc_calc = KM_COMS_crc8(msg.len, msg.type, msg.payload);
208 if(crc_calc == msg.crc) {
209 KM_COMS_ProccessPayload(msg);
210 }
211
212 processed += total_len;
213 }
214
215 // Compactar buffer: mover bytes no procesados al inicio
216 if(processed > 0 && processed < rx_buffer_len) {
217 memmove(rx_buffer, rx_buffer + processed, rx_buffer_len - processed);
218 rx_buffer_len -= processed;
219 } else if(processed >= rx_buffer_len) {
220 rx_buffer_len = 0;
221 }
222
223 xSemaphoreGive(km_coms_mutex);
224}

251static void KM_COMS_ProccessPayload(km_coms_msg msg) {
252 ESP_LOGI("KM_coms", "RX msg type=0x%02X len=%d crc=0x%02X", msg.type, msg.len, msg.crc);
253
254 int64_t object_value = 0;
255 int8_t direction;
256
257 switch (msg.type)
258 {
259 case ORIN_TARG_THROTTLE:
260 if (msg.len != 1) return; // Invalid payload
261
262 object_value = (int64_t)msg.payload[0];
263 KM_OBJ_SetObjectValue(TARGET_THROTTLE, object_value);
264 break;
265
266 case ORIN_TARG_BRAKING:
267 if (msg.len != 1) return; // Invalid payload
268 object_value = (int64_t)msg.payload[0];
269 KM_OBJ_SetObjectValue(TARGET_BRAKING, object_value);
270 break;
271
272 case ORIN_TARG_STEERING:
273 if (msg.len != 2) return; // Invalid payload
274 direction = msg.payload[0];
275 if (direction == 0) {
276 object_value = (int64_t)msg.payload[1];
277 } else if (direction == 1){
278 object_value -= (int64_t)msg.payload[1];
279 } else {
280 return; // Invalid payload
281 }
282
283 KM_OBJ_SetObjectValue(TARGET_STEERING, object_value);
284 break;
285
286 case ORIN_MISION:
287 if (msg.len != 1) return; // Invalid payload
288 object_value = (int64_t)msg.payload[0];
289 KM_OBJ_SetObjectValue(MISION_ORIN, object_value);
290 break;
291
292 case ORIN_MACHINE_STATE:
293 if (msg.len != 1) return; // Invalid payload
294 object_value = (int64_t)msg.payload[0];
295 KM_OBJ_SetObjectValue(MACHINE_STATE_ORIN, object_value);
296 break;
297
298 case ORIN_HEARTBEAT:
299 // Ns si guardarlo en la libreria de variables o reinicar algo. ns
300 break;
301
302 case ORIN_SHUTDOWN:
303 if (msg.len != 1) return; // Invalid payload
304 object_value = (int64_t)msg.payload[0];
305 KM_OBJ_SetObjectValue(SHUTDOWN_ORIN, object_value);
306 break;
307
308 case ORIN_COMPLETE:
309 if (msg.len != 7) return; // Invalid payload
310 object_value = (int64_t)msg.payload[0];
311 KM_OBJ_SetObjectValue(TARGET_THROTTLE, object_value);
312
313 object_value = (int64_t)msg.payload[1];
314 KM_OBJ_SetObjectValue(TARGET_BRAKING, object_value);
315
316 direction = msg.payload[2];
317 if (direction == 0) {
318 object_value = (int64_t)msg.payload[3];
319 } else if (direction == 1){
320 object_value -= (int64_t)msg.payload[3];
321 } else {
322 return; // Invalid payload
323 }
324 KM_OBJ_SetObjectValue(TARGET_STEERING, object_value);
325
326 object_value = (int64_t)msg.payload[4];
327 KM_OBJ_SetObjectValue(MISION_ORIN, object_value);
328
329 object_value = (int64_t)msg.payload[5];
330 KM_OBJ_SetObjectValue(MACHINE_STATE_ORIN, object_value);
331
332 object_value = (int64_t)msg.payload[6];
333 KM_OBJ_SetObjectValue(SHUTDOWN_ORIN, object_value);
334 break;
335
336 default:
337 break;
338 }

358static uint8_t KM_COMS_crc8(uint8_t len, uint8_t type, const uint8_t *data) {
359 uint8_t crc = 0x00;
360
361 crc ^= len;
362 for (int j = 0; j < 8; j++) {
363 crc = (crc & 0x80) ? (crc << 1) ^ 0x07 : (crc << 1);
364 }
365
366 crc ^= type;
367 for (int j = 0; j < 8; j++) {
368 crc = (crc & 0x80) ? (crc << 1) ^ 0x07 : (crc << 1);
369 }
370
371 for (uint8_t i = 0; i < len; i++) {
372 crc ^= data[i];
373 for (int j = 0; j < 8; j++) {
374 crc = (crc & 0x80) ? (crc << 1) ^ 0x07 : (crc << 1);
375 }
376 }
377
378 return crc;
379}
