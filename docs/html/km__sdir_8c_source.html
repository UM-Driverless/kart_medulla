61// Leer valor crudo (0-4095)
62uint16_t KM_SDIR_ReadRaw(sensor_struct *sensor) {
63 uint8_t data[2];
64
65 if (KM_SDIR_ReadRegisters(sensor, AS5600_ANGLE_MSB, data, 2)) {
66 sensor->lastRawValue = ((uint16_t)data[0] << 8) | data[1];
67 sensor->lastReadTime = esp_timer_get_time(); // microsegundos desde boot
68 sensor->errorCount = 0;
69 return sensor->lastRawValue;
70 } else {
71 sensor->errorCount++;
72 if (sensor->errorCount >= sensor->max_error_count) {
73 ESP_LOGW(TAG, "Error reading sensor, resetting I2C");
74 KM_SDIR_ResetI2C(sensor);
75 sensor->errorCount = 0;
76 }
77 return sensor->lastRawValue;
78 }
79}

131// Leer registros I2C
132static int8_t KM_SDIR_ReadRegisters(sensor_struct *sensor, uint8_t reg, uint8_t* data, uint8_t len) {
133 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
134 i2c_master_start(cmd);
135 i2c_master_write_byte(cmd, (AS5600_ADDR << 1) | I2C_MASTER_WRITE, true);
136 i2c_master_write_byte(cmd, reg, true);
137 i2c_master_stop(cmd);
138 esp_err_t ret = i2c_master_cmd_begin(I2C_NUM_0, cmd, pdMS_TO_TICKS(I2C_MASTER_TIMEOUT_MS));
139 i2c_cmd_link_delete(cmd);
140
141 if (ret != ESP_OK) {
142 return 0;
143 }
144
145 cmd = i2c_cmd_link_create();
146 i2c_master_start(cmd);
147 i2c_master_write_byte(cmd, (AS5600_ADDR << 1) | I2C_MASTER_READ, true);
148 if (len > 1) {
149 i2c_master_read(cmd, data, len - 1, I2C_MASTER_ACK);
150 }
151 i2c_master_read_byte(cmd, data + len - 1, I2C_MASTER_NACK);
152 i2c_master_stop(cmd);
153
154 ret = i2c_master_cmd_begin(I2C_NUM_0, cmd, pdMS_TO_TICKS(I2C_MASTER_TIMEOUT_MS));
155 i2c_cmd_link_delete(cmd);
156
157 return (ret == ESP_OK) ? 1 : 0;
158}

160// Devuelve ángulo en radianes (-PI a PI)
161static float KM_SDIR_ReadAngle(sensor_struct *sensor) {
162 uint16_t raw = KM_SDIR_ReadRaw(sensor);
163
164 int16_t centered = (int16_t)raw - sensor->centerOffset;
165 float angle = ((float)centered / (float)SENSOR_MAX) * 2.0f * MAX_RAD;
166
167 return angle;
168}
