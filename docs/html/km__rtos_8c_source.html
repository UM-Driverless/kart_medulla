78esp_err_t KM_RTOS_AddTask(RTOS_Task task){
79
80 // Buscar hueco libre en array y comprobar que no exista ya esa tarea
81 for (uint8_t i = 0; i < RTOS_MAX_TASKS; i++) {
82 // Hueco ocupado
83 if (tasks[i].handle != NULL) continue;
84
85 tasks[i] = task;
86 tasks[i].active = 1;
87
88 if (tasks[i].priority == 0) tasks[i].priority = RTOS_DEFAULT_PRIORITY;
89 if (tasks[i].stackSize == 0) tasks[i].stackSize = RTOS_DEFAULT_STACK_SIZE;
90 if (tasks[i].period_ms == 0) tasks[i].period_ms = RTOS_DEFAULT_PERIOD_MS;
91
92 BaseType_t result = xTaskCreate(
93 KM_RTOS_TaskWrapper, // Wrapper
94 task.name,
95 task.stackSize,
96 &tasks[i], // Pointer to this task
97 task.priority,
98 &tasks[i].handle
99 );
100
101 if (result != pdPASS) {
102 memset(&tasks[i], 0, sizeof(RTOS_Task));
103 return ESP_FAIL; // Error creating task
104 }
105
106 return ESP_OK; // Success
107 }
108
109 return ESP_FAIL; // Fail, no space in array
110}

243static void KM_RTOS_TaskWrapper(void *params) {
244 RTOS_Task *task = (RTOS_Task *)params;
245 TickType_t xLastWakeTime = xTaskGetTickCount();
246
247 if (!task || task->taskFn) vTaskDelete(NULL);
248
249 while (1) {
250 if (task->active && task->taskFn != NULL) {
251 task->taskFn(task->context); // Call the logical function
252 }
253
254 vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(task->period_ms));
255 }
256}
