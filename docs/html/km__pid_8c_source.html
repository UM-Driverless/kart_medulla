37float KM_PID_Calculate(PID_Controller *controller, float setpoint, float measurement) {
38 unsigned long currentTime = esp_timer_get_time();
39 float dt = (currentTime - controller->lastTime) / 1000000.0f; // Convert to seconds
40
41 // Avoid division by zero
42 if (dt <= 0.0f) {
43 dt = 0.001f; // Minimum 1ms
44 }
45
46 // Calculate error
47 float error = setpoint - measurement;
48
49 // Proportional term
50 float pTerm = controller->kp * error;
51
52 // Integral term with anti-windup
53 controller->integral += error * dt;
54
55 if (controller->integral < controller->integralMin) {
56 controller->integral = controller->integralMin;
57 }
58
59 if (controller->integral > controller->integralMax) {
60 controller->integral = controller->integralMax;
61 }
62
63 float iTerm = controller->ki * controller->integral;
64
65 // Derivative term
66 float derivative = (error - controller->lastError) / dt;
67 float dTerm = controller->kd * derivative;
68
69 // Calculate total output
70 float output = pTerm + iTerm + dTerm;
71
72 // Apply output limits
73 if (output < controller->outputMin) output = controller->outputMin;
74 if (output > controller->outputMax) output = controller->outputMax;
75
76
77 // Update state
78 controller->lastError = error;
79 controller->lastTime = currentTime;
80
81 return output;
82}
