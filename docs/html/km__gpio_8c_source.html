178/* ---------- Initialization ---------- */
179esp_err_t KM_GPIO_Init(void)
180{
181 esp_err_t ret;
182
183 // Enable DAC channels
184 ret = dac_output_enable(DAC_CHAN_0); // CMD_ACC
185 if (ret != ESP_OK) return ret;
186 ret = dac_output_enable(DAC_CHAN_1); // CMD_BRAKE
187 if (ret != ESP_OK) return ret;
188
189 // Setup PWM (LEDC)
190 ledc_timer_config_t pwm_timer = {
191 .speed_mode = LEDC_HIGH_SPEED_MODE,
192 .duty_resolution = LEDC_TIMER_8_BIT,
193 .timer_num = LEDC_TIMER_0,
194 .freq_hz = 1000,
195 .clk_cfg = LEDC_AUTO_CLK
196 };
197 ret = ledc_timer_config(&pwm_timer);
198 if (ret != ESP_OK) return ret;
199
200 ledc_channel_config_t pwm_channel = {
201 .gpio_num = (gpio_num_t)PIN_STEER_PWM,
202 .speed_mode = LEDC_HIGH_SPEED_MODE,
203 .channel = LEDC_CHANNEL_0,
204 .intr_type = LEDC_INTR_DISABLE,
205 .timer_sel = LEDC_TIMER_0,
206 .duty = 0
207 };
208 ret = ledc_channel_config(&pwm_channel);
209 if (ret != ESP_OK) return ret;
210
211 return ESP_OK;
212}

230/* ---------- ADC ---------- */
231uint16_t KM_GPIO_ReadADC(const gpio_config_t *pin)
232{
233 gpio_num_t gpio = (gpio_num_t)(pin->pin_bit_mask);
234 int raw_out_adc2 = 0;
235
236 switch (gpio)
237 {
238 case GPIO_NUM_36: return (uint16_t)adc1_get_raw(ADC1_CHANNEL_0); // pressure 1
239 case GPIO_NUM_39: return (uint16_t)adc1_get_raw(ADC1_CHANNEL_3); // pressure 2
240 case GPIO_NUM_34: return (uint16_t)adc1_get_raw(ADC1_CHANNEL_6); // pressure 3
241 case GPIO_NUM_35: return (uint16_t)adc1_get_raw(ADC1_CHANNEL_7); // pedal acc
242 case GPIO_NUM_32: return (uint16_t)adc1_get_raw(ADC1_CHANNEL_4); // pedal brake
243 case GPIO_NUM_33: return (uint16_t)adc1_get_raw(ADC1_CHANNEL_5); // hydraulic 1
244 case GPIO_NUM_13: // hydraulic 2
245 if (adc2_get_raw(ADC2_CHANNEL_4, ADC_WIDTH_BIT_12, &raw_out_adc2) == ESP_OK)
246 return raw_out_adc2;
247 return 0;
248
249 default: return 0;
250 }
251}

253/* ---------- DAC ---------- */
254esp_err_t KM_GPIO_WriteDAC(const gpio_config_t *pin, uint8_t value)
255{
256 gpio_num_t gpio = (gpio_num_t)(pin->pin_bit_mask);
257
258 if (gpio == PIN_CMD_ACC) return dac_output_voltage(DAC_CHAN_0, value);
259 if (gpio == PIN_CMD_BRAKE) return dac_output_voltage(DAC_CHAN_1, value);
260
261 return ESP_ERR_INVALID_ARG;
262}

272 return ledc_update_duty(LEDC_HIGH_SPEED_MODE, LEDC_CHANNEL_0);
